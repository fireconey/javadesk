可以在自己的函数中重写__str__方法
class Person(object):

    def __init__(self, name, gender):
        self.name = name
        self.gender = gender

class Student(Person):

    def __init__(self, name, gender, score):
        super(Student, self).__init__(name, gender)
        self.score = score

    def __str__(self):
        return "1111111"

s = Student('Bob', 'male', 88)
print (s)
